      username: "user2"
      password: "user2"
      first_name: "user"
      last_name: "2",
      date_of_birth: "2003-03-03",
      bio: "stu/dent",
    }
  ) {
    user_id
    username
    first_name
    last_name
    date_of_birth
    bio
    profile_picture
  }
}

# 2. Login user 2
mutation login_user_2 {
  login(username: "user2", password: "user2") {
    token
  }
}

# 3. Follow user 1 
mutation follow {
  createUserInteraction(userInteraction: {
    user_id_initiator: 2,
    user_id_recipient: 1,
    interaction_type: FOLLOW
  }) {
    user_interaction_id
    interaction_type
    user_id_initiator
    user_id_recipient
    timestamp
  }
}

# 4. Like user 1's post
mutation like_post {
  createPostInteraction(postInteraction: { 
    user_id: 2,
    post_id: 1,
    interaction_type: LIKE
  }) {
    post_interaction_id
    post_id
    user_id
    interaction_type
    timestamp
  }
}

# 7. Poke user 1 -> user 2
mutation poke {
  createUserInteraction(userInteraction: {

# 8. Join event (as user 1) 
mutation join_event {
